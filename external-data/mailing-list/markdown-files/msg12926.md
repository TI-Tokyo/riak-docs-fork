I do this in nodejs all the time using the riak-js package. Here is a modified 
map function that I use in production (i added o.fullreturn):

var map = function(v, keydata, args) {
 if (v.values) {
 var ret = [], o = {};
 o.values = Riak.mapValuesJson(v)[0]; 
 o.fullreturn = v;
 
 ret.push(o);
 return ret;
 } else {
 return [];
 }
};
